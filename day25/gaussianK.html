Define G matrix and Gaussian kernel functions

# G matrix
computeG <- function(X){
  X <- scale(X, center = TRUE, scale = TRUE)
  p <- ncol(X)
  G <- tcrossprod(X) / p
  return(G)
}

# Gaussian kernel matrix
computeGK <- function(X, bw){
  X <- scale(X, center = TRUE, scale = TRUE)
  D<- as.matrix(dist(X, method="euclidean"))^2
  D <- D/mean(D)
  GK <- exp(-bw * D) 
  return(GK)
}
